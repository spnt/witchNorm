TAB插件

TAB插件介绍

插件文件名为：witch.ui.tab.js，自行添加引用，实际使用多个插件时请合并

本插件是基于jQuery的插件
兼容性：IE7+
压缩后：820B
当前版本：0.1
默认参数：

  1. var defaults = {
  2.    activeClass:"f-tab-active",//默认标题选中样式
  3.    contentClass:'f-tab-c',//默认正文区样式
  4.    ajaxUrl:undefined,//通过ajax加载内容，此为ajax获取数据的地址，暂不支持JSONP//默认为不通过ajax
  5.    //若为ajax获取数据，则当正文区不为空时执行
  6.    //参数请写在响应的标题li上，属性为param，提交方法：POST
  7.     loaddingClass:'f-tab-loadding'//如果ajax，则loadding按钮样式
  8. }
